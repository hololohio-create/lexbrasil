ℹ️ Informações Técnicas

SQL para criar tabelas (executar no Supabase uma única vez):
CREATE TABLE novidades (id SERIAL PRIMARY KEY, tipo TEXT, fonte TEXT, titulo TEXT, descricao TEXT, impacto TEXT, data_aproximada TEXT, criado_em TIMESTAMPTZ DEFAULT NOW()); CREATE TABLE meta (chave TEXT PRIMARY KEY, valor TEXT); CREATE TABLE normas_extras (id SERIAL PRIMARY KEY, tipo TEXT, nome TEXT, sigla TEXT, numero TEXT, url TEXT, texto TEXT, analise TEXT, temas TEXT, tribunal TEXT, tese TEXT, criado_em TIMESTAMPTZ DEFAULT NOW()); CREATE TABLE colaboradores (id SERIAL PRIMARY KEY, nome TEXT, cargo TEXT, usuario TEXT UNIQUE, senha TEXT, ativo BOOLEAN DEFAULT TRUE, criado_em TIMESTAMPTZ DEFAULT NOW()); CREATE TABLE colabs_conteudo (id SERIAL PRIMARY KEY, colab_id INT, autor TEXT, cargo TEXT, titulo TEXT, tipo TEXT, texto TEXT, expira_em TIMESTAMPTZ, criado_em TIMESTAMPTZ DEFAULT NOW());
